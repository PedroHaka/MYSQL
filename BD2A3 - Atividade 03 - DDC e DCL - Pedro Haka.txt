CREATE SCHEMA IF NOT EXISTS `Atividade 03`  DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci ;

USE `Atividade 03`; 



CREATE TABLE IF NOT EXISTS `cliente` (

	`id_cliente` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'Código do Cliente',
	
	`nome` VARCHAR(45) NOT NULL COMMENT 'Nome do Cliente',
    
	`telefone` INT UNSIGNED NOT NULL COMMENT 'Número de Telefone do Cliente',
    
	`dataNasc` DATE NOT NULL COMMENT 'Data de Nascimento do Cliente',
    
	`cpf` VARCHAR(11) NOT NULL COMMENT 'Número do CPF',
    
	`endereço` VARCHAR(100) NOT NULL COMMENT 'Endereço do Cliente',
    
	UNIQUE INDEX `cpf_UNIQUE` (`cpf` ASC),
    
	PRIMARY KEY (`id_cliente`)

)

ENGINE=InnoDB;
    
 

CREATE TABLE IF NOT EXISTS `fornecedor` (

	`id_fornecedor` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'Código do Fornecedor',
	`empresa` VARCHAR(45) NOT NULL COMMENT 'Nome da empresa',
    
	`cnpj` VARCHAR(14) NOT NULL COMMENT 'Número de registro da empresa',
    
	`endereço` VARCHAR(100) NOT NULL COMMENT 'Endereço do Fornecedor',
    
	`telefone` INT UNSIGNED NOT NULL COMMENT 'Número de telefone do Fornecedor',
   
	`e-mail` VARCHAR(225) NOT NULL COMMENT 'Endereço Eletrônico do Fornecedor',
    
	UNIQUE INDEX `cnpj_UNIQUE` (`cnpj` ASC),
    
	PRIMARY KEY (`id_fornecedor`)

 )

ENGINE=InnoDB;



CREATE TABLE IF NOT EXISTS `produtos` (
	`id_produto` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'Código do Produto',
    
	`nome` VARCHAR(45) NOT NULL COMMENT 'Nome do produto',
    
	`quantidade` INT UNSIGNED NOT NULL COMMENT 'Quantidade',
    
	`id_fornecedor` INT UNSIGNED NOT NULL COMMENT 'Código do Fornecedor/Chave estrangeira',
    
	UNIQUE INDEX nome_UNIQUE (`nome` ASC),
    
	PRIMARY KEY (`id_produto`),
    
	INDEX `fk_fornecedor` (`id_fornecedor` ASC),
    
	FOREIGN KEY (`id_fornecedor`)
        
	REFERENCES `atividade 03`.`fornecedor` (`id_fornecedor`)
        
	ON DELETE NO ACTION 
        
	ON UPDATE NO ACTION

)  

ENGINE=INNODB;


CREATE TABLE IF NOT EXISTS `vendas` (
	`id_vendas` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'Código da Venda',
    
	`id_cliente` INT UNSIGNED NOT NULL COMMENT 'Código do Cliente/Chave estrangeira',
    
	`id_produtos` INT UNSIGNED NOT NULL COMMENT 'Código dos Produtos/Chave estrangeira',
    
	`quantidade` BIGINT UNSIGNED NOT NULL COMMENT 'Quantidade vendida',
    
	PRIMARY KEY (`id_vendas`),
    
	INDEX `fk_cliente` (`id_cliente` ASC),
    
	INDEX `fk_produtos` (`id_produtos` ASC),
		
	FOREIGN KEY (`id_cliente`)
        
	REFERENCES `atividade 03`.`cliente` (`id_cliente`),
        
	FOREIGN KEY (`id_produtos`)
        
	REFERENCES `atividade 03`.`produtos` (`id_produtos`)
        
	ON DELETE NO ACTION 
        
	ON UPDATE NO ACTION

)  

ENGINE=INNODB;
